% PDFs
\pgfmathdeclarefunction{dnormal}{3}{%
  \pgfmathparse{1/(sqrt(2*pi*#3))*exp(-((#1-#2)^2)/(2*#3))}%
}

\pgfmathdeclarefunction{dexp}{2}{%
  \pgfmathparse{#2*exp(-#2*#1))}%
}

\pgfmathdeclarefunction{pexp}{2}{%
  \pgfmathparse{1-exp(-#2*#1))}%
}

\pgfmathdeclarefunction{dlog}{3}{%
  \pgfmathparse{exp(-(#1-#2)/#3)/(#3*(1+exp(-(#1-#2)/#3))^2)}%
}

\pgfmathdeclarefunction{plog}{3}{%
  \pgfmathparse{1/(1+exp(-(#1-#2)/#3))}%
}

\pgfmathdeclarefunction{qlog}{3}{%
  \pgfmathparse{#2+#3*log(#1/(1-#1))}%
}
